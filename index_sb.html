一、商品信息收录代码

嵌入代码 
if(!_BFD.addItemFlag){		
	//判断商品是否下架、添加商品、点击推荐请求、上传浏览行为
	if(_store == 0||_price == 0||_del == true||_iid == '296393'||_iid == '299383'||_iid == '299381'||_iid == '288290'||_iid == '288291'||_iid == '288292'||BFD_Info.isgift == "true"){
		//商品下架
		_this.send(new $Core.inputs.RmItem(_iid));
	}else{
		//添加商品
		var _attr = new $Core.inputs.JObject();
		_attr.stk = parseInt(_store);
		var add_item = new $Core.inputs.AddItem(_iid);
		add_item.simg = _simg;
		add_item.img = _bimg;
		add_item.name = _name;
		add_item.prc = _price;
		add_item.mktp = _mprice;
		add_item.memp = _vprice1;
		add_item.vipp = _vprice2;
		add_item.url = _link;
		add_item.brd = _brand;
		add_item.cat = _catid;
		add_item.grp = _grid;
		add_item.tag = _tag;
		add_item.clr = _color;
		add_item.stk = _store;
		add_item.stkcnt = _store;
		add_item.attr = _attr.toString();
		
		_this.send(add_item);
		_BFD.addItemFlag = true;	
	}
}

详细参数需要在对接实施环节，与我方研发沟通后确认。
